Apa itu LiveScript?

LiveScript is a language which compiles to JavaScript. It has a straightforward mapping to JavaScript and allows you to write expressive code devoid of repetitive boilerplate. While LiveScript adds many features to assist in functional style programming, it also has many improvements for object oriented and imperative programming.

Bagaimana itu LiveScript? 

require! \http

server = http.createServer !(req, res) ->
  res.writeHead 200, do
    'Content-Type': 'text/plain'
  res.end 'Hello World\n'
server.listen 1337, '127.0.0.1'
console.log 'Server listening at http://127.0.0.1'
Become 

var http = require('http');

var server = http.createServer(function(req, res){
    res.writeHead(200, {'Content-Type': 'text/plain'});
    res.end('Hello World\n');
});
server.listen(1337, '127.0.0.1');
console.log('Server listening at http://127.0.0.1');

Switch-Case

in Javascript


var angka = 10;
switch (angka) {
case 10:
    console.log('ten');
    break;
case 5:
    console.log('five');
    break;
default:
    console.log('not five and ten')
}

in Coffeescript


angka = 10
switch angka
    when 10 then console.log 'ten'
    when 5 then console.log 'five'
    else console.log 'not five and ten'

in Livescript


angka = 10
switch angka
    when 10 then console.log 'ten'
    when 5 then console.log 'five'
    else console.log 'not five and ten'

switch angka
| 10    => console.log 'ten'
| 5     => console.log 'five'
| otherwise => console.log 'not five and ten'
